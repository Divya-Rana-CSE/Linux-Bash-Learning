ls -> List Storage files or folder inside that folder
pwd -> where we are in the directory
cd -> change directory
cd ~ -> change directory to home directory
cd /etc -> change directory to /etc folder
nano practice.sh -> create/edit a bash script file 
sudo chmod -x practice.sh -> change file's premission to executable(-x) using admin permission
ls -l -> List Storage in more detailed manner
./myscript.sh -> runs the script 
echo $SHELL -> which shell script command line using in the system
which bash -> where bash present 
cat myscript.sh -> used to view and work with text files 

#!/bin/bash -> Shebang (#!/bin/bash) tells the interpreter which language to use. The .sh extension is optional—it's just a naming convention. Even without .sh, the shebang forces it to run as a bash script.
